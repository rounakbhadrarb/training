package com.company;

import java.util.Scanner;

public class Main {

    public static void main(String[] args) {
	// write your code here
        int len=Integer.parseInt(args[0]);

        //declaration of array
        int[] arr=new int[len];
        Scanner sc=new Scanner(System.in);
        for(int i=0;i<arr.length;i++)
        {
            arr[i]=sc.nextInt();
        }
        for(int i=0;i<(arr.length-1);i++)
        {
            for(int j=0;j< (arr.length-i-1);j++)
            {
                if(arr[j]>arr[j+1])
                {
                    int temp=arr[j];
                    arr[j]=arr[j+1];
                    arr[j+1]=temp;
                }
            }
        }
        for(int i=0;i< arr.length;i++)
        {
            System.out.print(arr[i]+" ");
        }


    }
}

package com.company;

public class customer {
    private int custId;
    private String custName;
    private int unitsConsumed;
    private int unitPrice;

    public customer(int custId, String custName, int unitsConsumed, int unitPrice) {
        this.custId = custId;
        this.custName = custName;
        this.unitsConsumed = unitsConsumed;
        this.unitPrice = unitPrice;
    }
    private int calc_billamt()
    {
        return(unitsConsumed*unitPrice);
    }

    @Override
    public String toString() {
        return "customer{" +
                "custId=" + custId +
                ", custName='" + custName + '\'' +
                ", unitsConsumed=" + unitsConsumed +
                ", unitPrice=" + unitPrice +"Bill amount:"+calc_billamt()+
                '}';
    }

package com.company;

public class Student {
    private int std_id;
    private String name;
    private int sub1;
    private int sub2;
    private int sub3;

    public Student(int std_id, String name, int sub1, int sub2, int sub3) {
        this.std_id = std_id;
        this.name = name;
        this.sub1 = sub1;
        this.sub2 = sub2;
        this.sub3 = sub3;
    }
    private double calc_per()
    {
        return((sub1+sub2+sub3)/3);
    }

    @Override
    public String toString() {
        return "Student{" +
                "std_id=" + std_id +
                ", name='" + name + '\'' +
                ", sub1=" + sub1 +
                ", sub2=" + sub2 +
                ", sub3=" + sub3 +"percentage scored"+calc_per()+
                '}';
    }
}
